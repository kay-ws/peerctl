# peerctl Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** VS Code / MCP に依存せず、tmux + git worktree + Claude Code Stop hook + JSONL transcript だけで「1 オーケストレータ + 複数ピア」を回す単一 bash CLI `peerctl` を作る。

**Architecture:** ピア = git worktree（または素 dir）で隔離され tmux window で動く Claude Code セッション。spawn が各ピアに Stop hook を仕込み、ピアはターン完了時に `signal` ファイル（`epoch\tsession_id\ttranscript_path`）を上書きする。recv は送信時刻より新しい signal を待ち、transcript の最終 assistant ターンを抽出して返す。

**Tech Stack:** bash, jq, tmux, git, coreutils。テストは外部依存ゼロの自前 bash ハーネス。

---

## File Structure

- `peerctl` — メイン CLI。arg dispatch (`main`) + コマンド関数 (`cmd_*`) + 純粋ヘルパ (`extract_last_reply`, `meta_*`, `read_signal_*`, `render_hook` 等)。末尾に source-guard を置き、テストから source して関数単体を叩けるようにする。
- `tests/assert.sh` — 外部依存ゼロの assertion ヘルパ (`assert_eq`, `assert_contains`, `finish`)。
- `tests/fixtures/` — サンプル transcript JSONL。
- `tests/test_extract_reply.sh` — `extract_last_reply` の単体テスト。
- `tests/test_signal_meta.sh` — `meta_*` / `read_signal_*` の単体テスト。
- `tests/test_hook.sh` — `render_hook` が生成する on-stop スクリプトの挙動テスト。
- `tests/test_orchestration.sh` — stub エージェントで spawn/list/kill/recv を検証（実 claude 不要）。
- `tests/smoke.sh` — 実 claude を使う end-to-end スモーク（手動実行、tmux + claude 必須）。
- `README.md` — 使い方。

**規約（全スクリプト共通）:** shebang は `#!/bin/bash` → 空行 → `set -euo pipefail`。テストファイルは `peerctl` を source した直後に `set +eu +o pipefail` で errexit を解除する（assert が途中で落ちないように）。

**共通環境変数（peerctl が参照、テストで上書き可）:**

```bash
_pc_fail=0
assert_eq() {
  local exp="$1" act="$2" msg="${3:-eq}"
  if [[ "$exp" != "$act" ]]; then
    printf 'FAIL %s\n  expected: %q\n  actual:   %q\n' "$msg" "$exp" "$act" >&2
    _pc_fail=1
  else
    printf 'ok   %s\n' "$msg"

}
assert_contains() {
  local hay="$1" needle="$2" msg="${3:-contains}"
  if [[ "$hay" != *"$needle"* ]]; then
    printf 'FAIL %s\n  %q not found in %q\n' "$msg" "$needle" "$hay" >&2
    _pc_fail=1
  else
    printf 'ok   %s\n' "$msg"
  fi
}
finish() {
  if [[ $_pc_fail -eq 0 ]]; then echo "ALL PASS"; exit 0; else echo "FAILURES"; exit 1; fi
}
```

```bash
#!/bin/bash

set -euo pipefail

PEERCTL_HOME="${PEERCTL_HOME:-${XDG_RUNTIME_DIR:-/tmp}/peerctl}"
PEERCTL_TMUX_SESSION="${PEERCTL_TMUX_SESSION:-peers}"
PEERCTL_AGENT="${PEERCTL_AGENT:-claude}"

die() { printf 'peerctl: %s\n' "$*" >&2; exit 1; }

usage() {
  cat <<'EOF'
peerctl - orchestrate Claude Code peers over tmux

usage:
  peerctl spawn <name> [--base <ref>] [--no-git] [--dir <path>]
  peerctl send  <name> <message...>
  peerctl recv  <name> [--timeout <seconds>]
  peerctl ask   <name> <message...> [--timeout <seconds>]
  peerctl list
  peerctl kill  <name> [--keep-worktree]
EOF
}

main() {
  local cmd="${1:-}"; shift || true
  case "$cmd" in
    spawn) cmd_spawn "$@" ;;
    send)  cmd_send "$@" ;;
    recv)  cmd_recv "$@" ;;
    ask)   cmd_ask "$@" ;;
    list)  cmd_list "$@" ;;
    kill)  cmd_kill "$@" ;;
    ""|-h|--help|help) usage ;;
    *) usage >&2; die "unknown command: $cmd" ;;
  esac
}

if [[ "${BASH_SOURCE[0]}" == "${0}" ]]; then
  main "$@"
fi
```

```bash
# 最終 user 行より後ろの assistant テキストを連結して返す
extract_last_reply() {
  local file="$1"
  jq -rs '
    (map(.message.role // .type) | rindex("user")) as $u
    | if $u == null then "" else
        .[($u+1):]
        | map(select((.message.role // "") == "assistant") | .message.content)
        | map(if type == "array" then (map(select(.type == "text") | .text) | join("\n"))
              elif type == "string" then .
              else "" end)
        | map(select(. != ""))
        | join("\n")
      end
  ' "$file"
}
```

```bash
peer_dir() { printf '%s/peers/%s\n' "$PEERCTL_HOME" "$1"; }

# meta は key=value 行。値は read で安全に取り出す（source しない）
meta_set() {
  local pd="$1" key="$2" val="$3"
  mkdir -p "$pd"
  local tmp; tmp="$(mktemp)"
  { grep -v "^${key}=" "$pd/meta" 2>/dev/null || true; printf '%s=%s\n' "$key" "$val"; } > "$tmp"
  mv "$tmp" "$pd/meta"
}
meta_get() {
  local pd="$1" key="$2"
  [[ -f "$pd/meta" ]] || { printf ''; return 0; }
  local line; line="$(grep "^${key}=" "$pd/meta" | tail -1)"
  printf '%s' "${line#*=}"
}

read_signal_epoch() {
  local pd="$1"
  [[ -f "$pd/signal" ]] || { printf '0'; return 0; }
  cut -f1 "$pd/signal"
}
read_signal_field() {
  local pd="$1" n="$2"
  [[ -f "$pd/signal" ]] || { printf ''; return 0; }
  cut -f"$n" "$pd/signal"
}
```

```bash
# 指定 peerdir の signal に書き込む on-stop スクリプト本体を出力
render_hook() {
  local pd="$1"
  cat <<EOF
#!/bin/bash

input=\$(cat)
sid=\$(printf '%s' "\$input" | jq -r '.session_id // ""')
tp=\$(printf '%s' "\$input" | jq -r '.transcript_path // ""')
printf '%s\t%s\t%s\n' "\$(date +%s)" "\$sid" "\$tp" > "$pd/signal"
exit 0
EOF
}

# workdir に Stop hook 用の on-stop.sh と .claude/settings.local.json を設置
install_hook() {
  local workdir="$1" pd="$2"
  render_hook "$pd" > "$pd/on-stop.sh"
  chmod +x "$pd/on-stop.sh"
  mkdir -p "$workdir/.claude"
  cat > "$workdir/.claude/settings.local.json" <<EOF
{
  "hooks": {
    "Stop": [
      { "hooks": [ { "type": "command", "command": "$pd/on-stop.sh" } ] }
    ]
  }
}
EOF
}
```

```bash
_ensure_session() {
  tmux has-session -t "$PEERCTL_TMUX_SESSION" 2>/dev/null \
    || tmux new-session -d -s "$PEERCTL_TMUX_SESSION" -n __scratch
}

cmd_spawn() {
  local name="" base="HEAD" nogit=0 dir=""
  while [[ $# -gt 0 ]]; do
    case "$1" in
      --base) base="$2"; shift 2 ;;
      --no-git) nogit=1; shift ;;
      --dir) dir="$2"; shift 2 ;;
      -*) die "spawn: unknown option $1" ;;
      *) [[ -z "$name" ]] && name="$1" || die "spawn: extra arg $1"; shift ;;
    esac
  done
  [[ -n "$name" ]] || die "spawn: name required"
  local pd; pd="$(peer_dir "$name")"
  [[ -e "$pd/meta" ]] && die "spawn: peer '$name' already exists"

  local workdir branch=""
  if [[ -n "$dir" ]]; then
    workdir="$dir"; mkdir -p "$workdir"
  elif [[ $nogit -eq 1 ]]; then
    workdir="$(mktemp -d "${TMPDIR:-/tmp}/peer-$name.XXXX")"
  else
    git rev-parse --is-inside-work-tree >/dev/null 2>&1 || die "spawn: not in a git repo (use --no-git or --dir)"
    branch="peer/$name"
    workdir="$(git rev-parse --show-toplevel)/.peers/$name"
    git worktree add -b "$branch" "$workdir" "$base" >/dev/null
  fi

  mkdir -p "$pd"
  install_hook "$workdir" "$pd"
  meta_set "$pd" target "$PEERCTL_TMUX_SESSION:$name"
  meta_set "$pd" workdir "$workdir"
  meta_set "$pd" branch "$branch"

  _ensure_session
  tmux new-window -t "$PEERCTL_TMUX_SESSION" -n "$name" -c "$workdir"
  tmux send-keys -t "$PEERCTL_TMUX_SESSION:$name" -l -- "$PEERCTL_AGENT"
  tmux send-keys -t "$PEERCTL_TMUX_SESSION:$name" Enter
  printf 'spawned peer %s (workdir: %s)\n' "$name" "$workdir"
}

cmd_list() {
  local base="$PEERCTL_HOME/peers"
  [[ -d "$base" ]] || { echo "(no peers)"; return 0; }
  local pd name target sig last status
  for pd in "$base"/*/; do
    [[ -d "$pd" ]] || continue
    name="$(basename "$pd")"
    target="$(meta_get "$pd" target)"
    sig="$(read_signal_epoch "$pd")"
    last="$(cat "$pd/last_send" 2>/dev/null || echo 0)"
    if [[ "$last" -gt "$sig" ]]; then status="working"; else status="idle"; fi
    printf '%-16s %-20s %s\n' "$name" "$target" "$status"
  done
}

cmd_kill() {
  local name="" keep=0
  while [[ $# -gt 0 ]]; do
    case "$1" in
      --keep-worktree) keep=1; shift ;;
      *) [[ -z "$name" ]] && name="$1" || die "kill: extra arg $1"; shift ;;
    esac
  done
  [[ -n "$name" ]] || die "kill: name required"
  local pd; pd="$(peer_dir "$name")"
  local target workdir branch
  target="$(meta_get "$pd" target)"; workdir="$(meta_get "$pd" workdir)"; branch="$(meta_get "$pd" branch)"
  [[ -n "$target" ]] && tmux kill-window -t "$target" 2>/dev/null || true
  if [[ $keep -eq 0 && -n "$branch" && -n "$workdir" ]]; then
    git worktree remove --force "$workdir" 2>/dev/null || true
  fi
  rm -rf "$pd"
  printf 'killed peer %s\n' "$name"
}
```

```bash
cmd_send() {
  local name="${1:-}"; shift || true
  [[ -n "$name" ]] || die "send: name required"
  local msg="$*"
  [[ -n "$msg" ]] || die "send: message required"
  local pd; pd="$(peer_dir "$name")"
  local target; target="$(meta_get "$pd" target)"
  [[ -n "$target" ]] || die "send: unknown peer '$name'"
  date +%s > "$pd/last_send"
  # 改行を空白化 → 本文(リテラル, Enter なし) → 独立 Enter の 2 段送信
  local oneline="${msg//$'\n'/ }"
  tmux send-keys -t "$target" -l -- "$oneline"
  tmux send-keys -t "$target" Enter
}

cmd_recv() {
  local name="" timeout=120
  while [[ $# -gt 0 ]]; do
    case "$1" in
      --timeout) timeout="$2"; shift 2 ;;
      *) [[ -z "$name" ]] && name="$1" || die "recv: extra arg $1"; shift ;;
    esac
  done
  [[ -n "$name" ]] || die "recv: name required"
  local pd; pd="$(peer_dir "$name")"
  [[ -f "$pd/meta" ]] || die "recv: unknown peer '$name'"
  local since; since="$(cat "$pd/last_send" 2>/dev/null || echo 0)"
  local deadline=$(( $(date +%s) + timeout ))
  while [[ $(date +%s) -lt $deadline ]]; do
    local sig; sig="$(read_signal_epoch "$pd")"
    if [[ "$sig" -gt "$since" ]]; then
      local tp; tp="$(read_signal_field "$pd" 3)"
      [[ -f "$tp" ]] && { extract_last_reply "$tp"; return 0; }
    fi
    sleep 0.3
  done
  die "recv: timeout waiting for '$name' (peer may still be working)"
}

cmd_ask() {
  local name="${1:-}"; shift || true
  [[ -n "$name" ]] || die "ask: name required"
  local timeout=120 args=()
  while [[ $# -gt 0 ]]; do
    case "$1" in
      --timeout) timeout="$2"; shift 2 ;;
      *) args+=("$1"); shift ;;
    esac
  done
  cmd_send "$name" "${args[@]}"
  cmd_recv "$name" --timeout "$timeout"
}
```

# peerctl

VS Code も MCP も使わず、tmux + git worktree + Claude Code Stop hook + JSONL transcript だけで
「1 オーケストレータ + 複数ピア」の Claude Code 並列オーケストレーションを行う単一 bash CLI。

## 依存
- bash, jq, tmux, git, coreutils

## 使い方
\`\`\`bash
peerctl spawn alpha            # worktree + tmux window で claude ピアを起動
peerctl ask alpha "実装して"   # 送信して応答（ターン完了）を待って返す
peerctl list                   # ピア一覧と状態 (idle/working)
peerctl kill alpha             # ピアと worktree を撤収

# 送受信を分けたい場合
peerctl send alpha "調べて"
peerctl recv alpha --timeout 60

peerctl spawn talker --no-git

## 仕組み
- 送信: \`tmux send-keys\` で本文と Enter を別キーとして送る（TUI の bracketed paste で改行が

- 隔離: ピアごとに git worktree（\`--no-git\` で素ディレクトリ）。cwd が分かれるので
  transcript の名前空間も作業も衝突しない
